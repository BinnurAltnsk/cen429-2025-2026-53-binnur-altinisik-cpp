 435  rv = g_pFunctionList->C_GetSlotList(CK_TRUE, slots.data(), &slotCount);
 436
 437  if (rv != CKR_OK) {
 438    return pkcs11ToErrorCode(rv);
 439  }
 440
 441  /**
 442   * @brief Label'a göre token bulma
 443   *
 444   * Her slot'taki token'ın label'ı kontrol edilir ve aranan label ile
 445   * karşılaştırılır. Token label'ları PKCS#11'de null-terminated değildir,
 446   * boşluklarla doldurulmuş olabilir, bu nedenle temizlenir.
 447   */
 448  // Label'a göre token bul
 449  CK_SLOT_ID selectedSlot = slots[0];  /**< Varsayılan olarak ilk slot */
 450  const char *searchLabel = label ? label : g_tokenLabel;  /**< Aranacak token label */
 451
 452  /**
 453   * @brief Token label karşılaştırması
 454   *
 455   * Her slot'taki token'ın label'ı C_GetTokenInfo ile alınır ve
 456   * aranan label ile karşılaştırılır. Eşleşme bulunursa, o slot seçilir.
 457   */
 458  for (CK_ULONG i = 0; i < slotCount; ++i) {
 459    CK_TOKEN_INFO tokenInfo;
 460    rv = g_pFunctionList->C_GetTokenInfo(slots[i], &tokenInfo);
 461
 462    if (rv == CKR_OK) {
 463      // Label karşılaştırması (null-terminated değil, boşluklarla doldurulmuş)
 464      char tokenLabel[33] = { 0 };
 465      SafeString::safeCopy(tokenLabel, sizeof(tokenLabel), tokenInfo.label);
 466      // Son boşlukları temizle
 467      size_t len = std::strlen(tokenLabel);
 468
 469      while (len > 0 && tokenLabel[len - 1] == ' ') {
 470        tokenLabel[--len] = '\0';
 471      }
 472
 473      if (std::strcmp(tokenLabel, searchLabel) == 0) {
 474        selectedSlot = slots[i];
 475        break;
 476      }
 477    }
 478  }
 479
 480  /**
 481   * @brief PKCS#11 session açma
 482   *
 483   * Seçilen slot'ta read-write session açılır. CKF_RW_SESSION flag'i ile
 484   * yazma işlemleri yapılabilir, CKF_SERIAL_SESSION flag'i ile serial
 485   * session açılır (paralel session değil).
 486   */
 487  // Session aç
 488  CK_ULONG flags = CKF_RW_SESSION | CKF_SERIAL_SESSION;  /**< Read-write ve serial session flag'leri */
 489  rv = g_pFunctionList->C_OpenSession(selectedSlot, flags, nullptr, nullptr, &g_sessionHandle);
 490
 491  if (rv != CKR_OK) {
 492    return pkcs11ToErrorCode(rv);
 493  }
 494
 495  /**
 496   * @brief PIN ile login
 497   *
 498   * Kullanıcı PIN'i ile token'a login yapılır. CKU_USER ile normal kullanıcı
 499   * olarak login yapılır. Eğer zaten login yapılmışsa (CKR_USER_ALREADY_LOGGED_IN),
 500   * hata olarak kabul edilmez.
 501   */
 502  // PIN ile login yap
 503  const char *loginPin = pin ? pin : g_pin;  /**< Kullanılacak PIN (parametre veya kaydedilmiş) */
 504  rv = g_pFunctionList->C_Login(g_sessionHandle, CKU_USER,
 505                                reinterpret_cast<CK_BYTE *>(const_cast<char *>(loginPin)),
 506                                static_cast<CK_ULONG>(std::strlen(loginPin)));
 507
 508  if (rv != CKR_OK && rv != CKR_USER_ALREADY_LOGGED_IN) {
 509    g_pFunctionList->C_CloseSession(g_sessionHandle);
 510    g_sessionHandle = 0;
 511    return pkcs11ToErrorCode(rv);
 512  }
 513
 514  g_currentSlot = selectedSlot;
 515  return ErrorCode::Success;
 516}
 517
 518/**
 519 * @brief Token kapat ve session'ı sonlandır
 520 *
 521 * Açık token session'ını kapatır. Önce logout yapılır, sonra session kapatılır.
 522 * Session handle ve current slot sıfırlanır.
 523 *
 524 * @note Bu fonksiyon, session açık değilse hiçbir işlem yapmaz ve Success döner.
 525 *
 526 * @return ErrorCode Başarı durumu
 527 */
 1528ErrorCode closeToken() {
 529  if (g_sessionHandle == 0 || !g_pFunctionList) {
 530    return ErrorCode::Success;
 531  }
 532
 533  g_pFunctionList->C_Logout(g_sessionHandle);
 534  g_pFunctionList->C_CloseSession(g_sessionHandle);
 535  g_sessionHandle = 0;
 536  g_currentSlot = 0;
 537  return ErrorCode::Success;
 538}
 539
 540/**
 541 * @brief Kriptografik anahtar oluştur
 542 *
 543 * Belirtilen tipte ve kullanım amacına göre kriptografik anahtar oluşturur.
 544 * Desteklenen anahtar tipleri: AES-256, RSA-2048, EC-P256.
 545 * Anahtar, token üzerinde saklanır ve keyId olarak döndürülür.
 546 *
 547 * @note Bu fonksiyon, açık bir session gerektirir (openToken() çağrılmış olmalı).
 548 * Simetrik anahtarlar (AES) için C_GenerateKey, asimetrik anahtarlar (RSA, EC)
 549 * için C_GenerateKeyPair kullanılır.
 550 *
 551 * @param keyType Anahtar tipi (AES_256, RSA_2048, EC_P256)
 552 * @param keyUsage Anahtar kullanım amacı (ENCRYPT_DECRYPT, SIGN_VERIFY, vb.)
 553 * @param keyLabel Anahtar etiketi (nullptr ise InvalidInput döner)
 554 * @param keyId Oluşturulan anahtarın ID'si (çıktı, nullptr ise InvalidInput döner)
 555 * @param keyIdLen keyId buffer boyutu (giriş), oluşturulan ID boyutu (çıktı)
 556 * @return ErrorCode Başarı durumu
 557 */
 1558ErrorCode generateKey(KeyType keyType, KeyUsage keyUsage,
 559                      const char *keyLabel, uint8_t *keyId, size_t &keyIdLen) {
 560  if (g_sessionHandle == 0 || !g_pFunctionList || !keyLabel || !keyId) {
 561    return ErrorCode::InvalidInput;
 562  }
 563
 564  /**
 565   * @brief PKCS#11 mekanizma ve attribute template oluşturma
 566   *
 567   * Key type'a göre uygun PKCS#11 mekanizması ve attribute template'leri
 568   * oluşturulur. Her anahtar tipi için farklı mekanizma ve key class kullanılır.
 569   */
 570  CK_MECHANISM mechanism = { 0 };
 571  CK_ATTRIBUTE template_[10];
 572  CK_ULONG templateCount = 0;
 573  // Key type'a göre mechanism ve key class ayarla
 574  CK_ULONG keyClass = CKO_SECRET_KEY;  /**< Varsayılan: Secret key (AES için) */
 575  CK_ULONG ckKeyType = CKK_AES;        /**< Varsayılan: AES key type */
 576  CK_ULONG keySize = 256 / 8;          /**< 32 bytes for AES-256 */
 577
 578  /**
 579   * @brief Anahtar tipine göre PKCS#11 mekanizması ve key class ayarlama
 580   *
 581   * Her anahtar tipi için farklı PKCS#11 mekanizması ve key class kullanılır:
 582   * - AES_256: CKM_AES_KEY_GEN, CKO_SECRET_KEY, CKK_AES
 583   * - RSA_2048: CKM_RSA_PKCS_KEY_PAIR_GEN, CKO_PRIVATE_KEY, CKK_RSA
 584   * - EC_P256: CKM_EC_KEY_PAIR_GEN, CKO_PRIVATE_KEY, CKK_EC
 585   */
 586  switch (keyType) {
 587    case KeyType::AES_256:
 588      mechanism.mechanism = CKM_AES_KEY_GEN;
 589      ckKeyType = CKK_AES;
 590      keySize = 32;  // 256 bits = 32 bytes
 591      break;
 592
 593    case KeyType::RSA_2048:
 594      keyClass = CKO_PRIVATE_KEY;
 595      mechanism.mechanism = CKM_RSA_PKCS_KEY_PAIR_GEN;
 596      ckKeyType = CKK_RSA;
 597      keySize = 0;  // RSA için farklı
 598      break;
 599
 600    case KeyType::EC_P256:
 601      keyClass = CKO_PRIVATE_KEY;
 602      mechanism.mechanism = CKM_EC_KEY_PAIR_GEN;
 603      ckKeyType = CKK_EC;
 604      keySize = 0;
 605      break;
 606  }
 607
 608  /**
 609   * @brief PKCS#11 attribute template oluşturma
 610   *
 611   * Anahtar özelliklerini belirten PKCS#11 attribute'ları oluşturulur.
 612   * Her attribute, anahtarın nasıl kullanılacağını ve saklanacağını belirtir.
 613   */
 614  // Template oluştur
 615  template_[templateCount].type = CKA_CLASS;        /**< Anahtar sınıfı (secret key veya private key) */
 616  template_[templateCount].pValue = &keyClass;
 617  template_[templateCount].ulValueLen = sizeof(keyClass);
 618  templateCount++;
 619  template_[templateCount].type = CKA_KEY_TYPE;     /**< Anahtar tipi (AES, RSA, EC) */
 620  template_[templateCount].pValue = &ckKeyType;
 621  template_[templateCount].ulValueLen = sizeof(ckKeyType);
 622  templateCount++;
 623  template_[templateCount].type = CKA_LABEL;       /**< Anahtar etiketi (kullanıcı tanımlı) */
 624  template_[templateCount].pValue = const_cast<char *>(keyLabel);
 625  template_[templateCount].ulValueLen = static_cast<CK_ULONG>(std::strlen(keyLabel));
 626  templateCount++;
 627  CK_BYTE true_value = CK_TRUE;   /**< Boolean true değeri (PKCS#11 attribute'ları için) */
 628  CK_BYTE false_value = CK_FALSE; /**< Boolean false değeri (kullanılmıyor) */
 629  template_[templateCount].type = CKA_TOKEN;       /**< Anahtar token'da saklanacak (kalıcı) */
 630  template_[templateCount].pValue = &true_value;
 631  template_[templateCount].ulValueLen = sizeof(true_value);
 632  templateCount++;
 633  template_[templateCount].type = CKA_PRIVATE;      /**< Anahtar private (sadece PIN ile erişilebilir) */
 634  template_[templateCount].pValue = &true_value;
 635  template_[templateCount].ulValueLen = sizeof(true_value);
 636  templateCount++;
 637
 638  /**
 639   * @brief Key usage attributes - Anahtar kullanım amacına göre özellikler
 640   *
 641   * Anahtarın hangi işlemler için kullanılabileceğini belirten attribute'lar eklenir:
 642   * - ENCRYPT_DECRYPT: CKA_ENCRYPT ve CKA_DECRYPT attribute'ları
 643   * - SIGN_VERIFY: CKA_SIGN ve CKA_VERIFY attribute'ları
 644   */
 645  // Key usage attributes
 646  if (keyUsage == KeyUsage::ENCRYPT_DECRYPT) {
 647    template_[templateCount].type = CKA_ENCRYPT;
 648    template_[templateCount].pValue = &true_value;
 649    template_[templateCount].ulValueLen = sizeof(true_value);
 650    templateCount++;
 651    template_[templateCount].type = CKA_DECRYPT;
 652    template_[templateCount].pValue = &true_value;
 653    template_[templateCount].ulValueLen = sizeof(true_value);
 654    templateCount++;
 655  } else if (keyUsage == KeyUsage::SIGN_VERIFY) {
 656    template_[templateCount].type = CKA_SIGN;
 657    template_[templateCount].pValue = &true_value;
 658    template_[templateCount].ulValueLen = sizeof(true_value);
 659    templateCount++;
 660    template_[templateCount].type = CKA_VERIFY;
 661    template_[templateCount].pValue = &true_value;
 662    template_[templateCount].ulValueLen = sizeof(true_value);
 663    templateCount++;
 664  }
 665
 666  /**
 667   * @brief AES için key size attribute'u
 668   *
 669   * AES anahtarları için key size (256 bit = 32 byte) belirtilir.
 670   * RSA ve EC anahtarları için key size attribute'u gerekmez.
 671   */
 672  // AES için key size
 673  if (keyType == KeyType::AES_256 && keySize > 0) {
 674    template_[templateCount].type = CKA_VALUE_LEN;  /**< AES anahtar uzunluğu (32 byte) */
 675    template_[templateCount].pValue = &keySize;
 676    template_[templateCount].ulValueLen = sizeof(keySize);
 677    templateCount++;
 678  }
 679
 680  /**
 681   * @brief PKCS#11 anahtar oluşturma
 682   *
 683   * Simetrik anahtarlar (AES) için C_GenerateKey, asimetrik anahtarlar (RSA, EC)
 684   * için C_GenerateKeyPair kullanılır. Key pair generation için public key
 685   * template'i de oluşturulmalıdır.
 686   */
 687  // Key generate
 688  CK_OBJECT_HANDLE keyHandle = 0;  /**< Oluşturulan anahtar handle'ı */
 689  CK_ULONG rv;                    /**< PKCS#11 return code */
 690
 691  if (keyType == KeyType::RSA_2048 || keyType == KeyType::EC_P256) {
 692    /**
 693     * @brief Key pair generation - Public key template
 694     *
 695     * Asimetrik anahtarlar (RSA, EC) için public key template'i oluşturulur.
 696     * Public key template'i, key class ve key type bilgilerini içerir.
 697     */
 698    // Key pair generation
 699    CK_ATTRIBUTE pubTemplate[5];  /**< Public key attribute template */
 700    CK_ULONG pubCount = 0;        /**< Public key template attribute sayısı */
 701    pubTemplate[pubCount].type = CKA_CLASS;
 702    pubTemplate[pubCount].pValue = &keyClass;
 703    pubTemplate[pubCount].ulValueLen = sizeof(keyClass);
 704    pubCount++;
 705    pubTemplate[pubCount].type = CKA_KEY_TYPE;
 706    pubTemplate[pubCount].pValue = &ckKeyType;
 707    pubTemplate[pubCount].ulValueLen = sizeof(ckKeyType);
 708    pubCount++;
 709    pubTemplate[pubCount].type = CKA_LABEL;
 710    pubTemplate[pubCount].pValue = const_cast<char *>(keyLabel);
 711    pubTemplate[pubCount].ulValueLen = static_cast<CK_ULONG>(std::strlen(keyLabel));
 712    pubCount++;
 713    CK_OBJECT_HANDLE pubKeyHandle = 0;
 714    rv = g_pFunctionList->C_GenerateKeyPair(g_sessionHandle, &mechanism,
 715                                            pubTemplate, pubCount,
 716                                            template_, templateCount,
 717                                            &pubKeyHandle, &keyHandle);
 718  } else {
 719    // Secret key generation
 720    rv = g_pFunctionList->C_GenerateKey(g_sessionHandle, &mechanism,
 721                                        template_, templateCount, &keyHandle);
 722  }
 723
 724  if (rv != CKR_OK) {
 725    return pkcs11ToErrorCode(rv);
 726  }
 727
 728  /**
 729   * @brief Key ID oluşturma
 730   *
 731   * Oluşturulan anahtarın handle'ı keyId olarak kullanılır.
 732   * Eğer keyId buffer yeterli büyüklükteyse (CK_OBJECT_HANDLE boyutu),
 733   * handle kopyalanır. Aksi takdirde, label'den hash oluşturulur.
 734   */
 735  // Key ID olarak handle'ı kullan (veya label'den hash)
 736  if (keyIdLen >= sizeof(CK_OBJECT_HANDLE)) {
 737    /** Key ID buffer yeterli: handle'ı kopyala */
 738    std::memcpy(keyId, &keyHandle, sizeof(CK_OBJECT_HANDLE));
 739    keyIdLen = sizeof(CK_OBJECT_HANDLE);
 740  } else {
 741    /**
 742     * Key ID buffer küçük: label'den hash oluştur
 743     *
 744     * Bu durumda, label'ın ilk N karakteri keyId olarak kullanılır.
 745     * N, keyId buffer boyutu ve label uzunluğunun minimum'udur.
 746     */
 747    // Key ID çok küçükse label'den hash oluştur
 748    size_t labelLen = std::strlen(keyLabel);
 749    size_t copyLen = (labelLen < keyIdLen) ? labelLen : keyIdLen;
 750    std::memcpy(keyId, keyLabel, copyLen);
 751    keyIdLen = copyLen;
 752  }
 753
 754  return ErrorCode::Success;
 755}
 756
 757/**
 758 * @brief Anahtarı label'a göre bul
 759 *
 760 * Belirtilen label'a sahip anahtarı token üzerinde arar ve keyId'sini döndürür.
 761 * Arama, C_FindObjectsInit ve C_FindObjects kullanılarak yapılır.
 762 *
 763 * @note Bu fonksiyon, açık bir session gerektirir (openToken() çağrılmış olmalı).
 764 * Anahtar bulunamazsa FileNotFound döner.
 765 *
 766 * @param keyLabel Anahtar etiketi (nullptr ise InvalidInput döner)
 767 * @param keyId Bulunan anahtarın ID'si (çıktı, nullptr ise InvalidInput döner)
 768 * @param keyIdLen keyId buffer boyutu (giriş), bulunan ID boyutu (çıktı)
 769 * @return ErrorCode Başarı durumu (Success, FileNotFound, InvalidInput)
 770 */
 1771ErrorCode findKey(const char *keyLabel, uint8_t *keyId, size_t &keyIdLen) {
 772  if (g_sessionHandle == 0 || !g_pFunctionList || !keyLabel || !keyId) {
 773    return ErrorCode::InvalidInput;
 774  }
 775
 776  // Template ile arama
 777  CK_ATTRIBUTE template_[3];
 778  template_[0].type = CKA_CLASS;
 779  CK_ULONG keyClass = CKO_SECRET_KEY;
 780  template_[0].pValue = &keyClass;
 781  template_[0].ulValueLen = sizeof(keyClass);
 782  template_[1].type = CKA_KEY_TYPE;
 783  CK_ULONG keyType = CKK_AES;
 784  template_[1].pValue = &keyType;
 785  template_[1].ulValueLen = sizeof(keyType);
 786  template_[2].type = CKA_LABEL;
 787  template_[2].pValue = const_cast<char *>(keyLabel);
 788  template_[2].ulValueLen = static_cast<CK_ULONG>(std::strlen(keyLabel));
 789  CK_ULONG rv = g_pFunctionList->C_FindObjectsInit(g_sessionHandle, template_, 3);
 790
 791  if (rv != CKR_OK) {
 792    return pkcs11ToErrorCode(rv);
 793  }
 794
 795  CK_OBJECT_HANDLE keyHandle = 0;
 796  CK_ULONG foundCount = 0;
 797  rv = g_pFunctionList->C_FindObjects(g_sessionHandle, &keyHandle, 1, &foundCount);
 798  g_pFunctionList->C_FindObjectsFinal(g_sessionHandle);
 799
 800  if (rv != CKR_OK || foundCount == 0) {
 801    return ErrorCode::FileNotFound;
 802  }
 803
 804  // Key ID olarak handle'ı kullan
 805  if (keyIdLen >= sizeof(CK_OBJECT_HANDLE)) {
 806    std::memcpy(keyId, &keyHandle, sizeof(CK_OBJECT_HANDLE));
 807    keyIdLen = sizeof(CK_OBJECT_HANDLE);
 808  } else {
 809    return ErrorCode::InvalidInput;
 810  }
 811
 812  return ErrorCode::Success;
 813}

 815/**
 816 * @brief Anahtarı sil
 817 *
 818 * Belirtilen keyId'ye sahip anahtarı token üzerinden siler.
 819 * C_DestroyObject kullanılarak anahtar kalıcı olarak silinir.
